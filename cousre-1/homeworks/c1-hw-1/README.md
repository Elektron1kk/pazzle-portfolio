git init - иницилизация нового репозитория в каталоге 
git status - статус выбранного каталога 
git add . - добавление всех новых и изменённых файлов 
git commit -m "ur commit" - коментарий к git add 
git rm file name - удаление файла из каталога 
git branch - перечисляет, создает или удаляет ветки 
git branch 'branch name' - создает новую ветку 
git checkout 'branch name' - Переключается на другую ветку 
git checkout -b 'branch name' - Создает новую ветку и сразу же переключается на нее 
git merge 'branch name' - Объединяет указанную ветку в текущую 
git pull - скачать изменения из удалённого репозитория 
git push - загружает на удалённый репозиторий изменения 
git log - история коммитов 
mkdir - создание папки 
cd 'ur directory' - смена директории 
cd .. - спуск на 1 ступень вниз
